<script>
  const form = document.getElementById('bookingForm');
  const summary = document.getElementById('summary');
  const price = document.getElementById('price');
  const toast = document.getElementById('toast');
  const dateInput = document.getElementById('date');
  const featureText = document.getElementById('featureText');

  const fields = ['name', 'service', 'date', 'time', 'payment', 'promo'];

  const features = [
    'Fast, flexible, and secure scheduling.',
    'Choose add-ons and get instant price updates.',
    'Promo codes can reduce your total automatically.',
    'Payment options make booking easier for everyone.'
  ];

  let featureIndex = 0;
  setInterval(() => {
    featureIndex = (featureIndex + 1) % features.length;
    featureText.textContent = features[featureIndex];
  }, 2500);

  function getBasePrice() {
    return parseInt(document.getElementById('service').value || '0', 10);
  }

  function getAddonPrice() {
    return [...document.querySelectorAll('.addon:checked')]
      .reduce((sum, el) => sum + parseInt(el.value, 10), 0);
  }

  function getDiscount(total) {
    const promo = document.getElementById('promo').value.trim().toUpperCase();
    if (promo === 'SAVE10') return total * 0.1;
    if (promo === 'SAVE20') return total * 0.2;
    return 0;
  }

  function updateSummary() {
    const name = document.getElementById('name').value || 'Guest';
    const serviceText = document.getElementById('service').selectedOptions[0]?.textContent || 'Service';
    const date = document.getElementById('date').value || 'Date';
    const time = document.getElementById('time').value || 'Time';
    summary.textContent = `${name} • ${serviceText} • ${date} • ${time}`;

    const base = getBasePrice();
    const addons = getAddonPrice();
    const discount = getDiscount(base + addons);
    const total = Math.max(0, base + addons - discount);

    price.textContent = `Total Price: $${total.toFixed(2)}${discount > 0 ? ` (Discount: $${discount.toFixed(2)})` : ''}`;
  }

  function setMinDate() {
    const today = new Date();
    const yyyy = today.getFullYear();
    const mm = String(today.getMonth() + 1).padStart(2, '0');
    const dd = String(today.getDate()).padStart(2, '0');
    dateInput.min = `${yyyy}-${mm}-${dd}`;
  }

  fields.forEach(id => {
    document.getElementById(id).addEventListener('input', updateSummary);
    document.getElementById(id).addEventListener('change', updateSummary);
  });

  document.querySelectorAll('.addon').forEach(cb => {
    cb.addEventListener('change', updateSummary);
  });

  dateInput.addEventListener('change', function () {
    if (this.value && this.value < this.min) {
      this.value = this.min;
    }
    updateSummary();
  });

  form.addEventListener('submit', function (e) {
    e.preventDefault();
    toast.classList.add('show');
    updateSummary();
    setTimeout(() => toast.classList.remove('show'), 2500);
  });

  form.addEventListener('reset', function () {
    setTimeout(() => {
      summary.textContent = 'No booking selected yet.';
      price.textContent = 'Total Price: $0';
      toast.classList.remove('show');
      updateSummary();
    }, 0);
  });

  setMinDate();
  updateSummary();
</script>
